Score Ring (pojok kanan atas)
Angka 78 = verdict.score — dihitung dari computeVerdict() (lihat Bagian 3).

import Svg, { Circle, Text as SvgText } from "react-native-svg";
function ScoreRing({ score, color }) {
  const r = 22, c = 2 * Math.PI * r;
  const dash = (score / 100) * c;
  return (
    <Svg width={60} height={60} viewBox="0 0 60 60">
      <Circle cx={30} cy={30} r={r} fill="none" stroke={color + "30"} strokeWidth={5} />
      <Circle cx={30} cy={30} r={r} fill="none" stroke={color} strokeWidth={5}
        strokeDasharray={`${dash} ${c}`} strokeLinecap="round"
        transform="rotate(-90, 30, 30)" />
      <SvgText x={30} y={35} textAnchor="middle" fontSize={13} fontWeight="bold" fill={color}>
        {score}
      </SvgText>
    </Svg>
  );
}

BAGIAN 3: VERDICT CARD
7 Kemungkinan Verdict
Verdict	Kondisi	Warna	Emoji	Score
STRONG BUY	isBuy + RSI < 35 + accuCount ≥ 2	Hijau	🚀	92
BUY ON WEAKNESS	isBuy + RSI < 35	Hijau	✅	78
ACCUMULATE	isBuy + tape = Accumulation	Hijau	⚡	72
BUY	isBuy (default)	Hijau	📈	65
STRONG SELL	isSell + RSI > 68 + distCount ≥ 2	Merah	🔴	88
AVOID / SELL	isSell	Merah	⚠️	60
WAIT & WATCH	semua kondisi lainnya	Amber	👀	45
function computeVerdict(analysis) {
  const sig        = (analysis?.overallSignal ?? "").toUpperCase();
  const rsi        = analysis?.indicators?.rsi ?? 50;
  const tape       = analysis?.tapeReadingSignal ?? "";
  const periods    = analysis?.periods ?? [];
  const isBuy      = sig.includes("BUY");
  const isSell     = sig.includes("SELL");
  const oversold   = rsi < 35;
  const overbought = rsi > 68;
  const accuCount  = periods.filter(p => p.signal.toUpperCase().includes("ACCUM")).length;
  const distCount  = periods.filter(p => p.signal.toUpperCase().includes("DIST")).length;
  if (isBuy && oversold && accuCount >= 2)
    return { label: "STRONG BUY",    sub: "Momentum balik — Konfirmasi teknikal kuat",
             color: "#34d399", emoji: "🚀", score: 92,
             action: "Segera akumulasi di zona support" };
  if (isBuy && oversold)
    return { label: "BUY ON WEAKNESS", sub: "RSI oversold — Peluang beli murah",
             color: "#34d399", emoji: "✅", score: 78,
             action: "Akumulasi bertahap di support zone" };
  if (isBuy && tape.toUpperCase().includes("ACCUM"))
    return { label: "ACCUMULATE",    sub: "Smart money masuk — potensi breakout",
             color: "#34d399", emoji: "⚡", score: 72,
             action: "Entry bertahap, konfirmasi volume" };
  if (isBuy)
    return { label: "BUY",           sub: "Sinyal teknikal positif",
             color: "#34d399", emoji: "📈", score: 65,
             action: "Entry di zona support dengan volume konfirmasi" };
  if (isSell && overbought && distCount >= 2)
    return { label: "STRONG SELL",   sub: "Distribusi aktif — hindari posisi baru",
             color: "#f87171", emoji: "🔴", score: 88,
             action: "Segera kurangi / exit posisi" };
  if (isSell)
    return { label: "AVOID / SELL",  sub: "Tekanan jual dominan saat ini",
             color: "#f87171", emoji: "⚠️", score: 60,
             action: "Tunggu reversal — jangan beli" };
  return   { label: "WAIT & WATCH",  sub: "Belum ada sinyal kuat — pantau dulu",
             color: "#fbbf24", emoji: "👀", score: 45,
             action: "Pasang alert di support, tunggu konfirmasi" };
}

Signal Badge (bar hijau di atas TP1/SL)
// "🔰 Data: Buy on Weakness · Grade D · hari ke-1"
function SignalBadge({ bowRow, bosRow }) {
  const source = bowRow ? "Buy on Weakness" : bosRow ? "Buy on Strength" : null;
  const grade  = bowRow?.grade ?? null;
  const days   = bowRow?.days ?? bosRow?.hold ?? null;
  if (!source) return null;
  
  const gradeColor = {
    "A": "#34d399", "B": "#60a5fa", "C": "#fbbf24", "D": "#f87171"
  }[grade] ?? "#94a3b8";
  
  return (
    <View style={{ flexDirection: "row", alignItems: "center", gap: 6, padding: 8,
      borderRadius: 10, backgroundColor: "#34d39920", borderWidth: 1, borderColor: "#34d39940" }}>
      <Text style={{ fontSize: 12 }}>🔰</Text>
      <Text style={{ color: "#34d399", fontWeight: "700", fontSize: 11 }}>
        Data: {source}
      </Text>
      {grade && (
        <View style={{ paddingHorizontal: 6, paddingVertical: 1, borderRadius: 6,
          backgroundColor: gradeColor + "25", borderWidth: 1, borderColor: gradeColor + "50" }}>
          <Text style={{ color: gradeColor, fontWeight: "900", fontSize: 10 }}>Grade {grade}</Text>
        </View>
      )}
      {days && <Text style={{ color: "#94a3b8", fontSize: 10 }}>· {days}</Text>}
    </View>
  );
}

function PlanBoxes({ entry, tp1, sl, tp2, rr }) {
  const tpPct  = entry > 0 ? ((tp1  - entry) / entry * 100).toFixed(1) : null;
  const slPct  = entry > 0 ? ((entry - sl)   / entry * 100).toFixed(1) : null;
  const tp2Pct = entry > 0 && tp2 > 0 ? ((tp2 - entry) / entry * 100).toFixed(1) : null;
  
  return (
    <View>
      {/* 2 kotak besar TP1 + SL */}
      <View style={{ flexDirection: "row", gap: 12 }}>
        <View style={{ flex: 1, padding: 16, borderRadius: 16,
          backgroundColor: "#34d39910", borderWidth: 2, borderColor: "#34d39940" }}>
          <Text style={{ color: "#94a3b8", fontSize: 9, fontWeight: "700" }}>🎯 TARGET (TP1)</Text>
          <Text style={{ color: "#34d399", fontWeight: "900", fontSize: 28, marginTop: 4 }}>
            {tp1?.toLocaleString("id-ID")}
          </Text>
          {tpPct && <Text style={{ color: "#34d399", fontSize: 13 }}>+{tpPct}%</Text>}
        </View>
        <View style={{ flex: 1, padding: 16, borderRadius: 16,
          backgroundColor: "#f8717110", borderWidth: 2, borderColor: "#f8717140" }}>
          <Text style={{ color: "#94a3b8", fontSize: 9, fontWeight: "700" }}>🔴 STOP LOSS</Text>
          <Text style={{ color: "#f87171", fontWeight: "900", fontSize: 28, marginTop: 4 }}>
            {sl?.toLocaleString("id-ID")}
          </Text>
          {slPct && <Text style={{ color: "#f87171", fontSize: 13 }}>-{slPct}%</Text>}
        </View>
      </View>
      
      {/* Baris ENTRY · TP2 · RSI · STOCHK */}
      <View style={{ flexDirection: "row", justifyContent: "space-between",
        marginTop: 12, padding: 12, borderRadius: 12, backgroundColor: "#1e2433" }}>
        <InfoCell label="ENTRY"  value={entry > 0 ? `${entry}–${Math.round(entry * 1.01)}` : "—"} color="#fbbf24" />

            <Text style={{ color: "#94a3b8", fontSize: 11 }}>Risk / Reward</Text>

// Helper bullet item
function BulletItem({ ok, text }) {
  const color = ok === "warn" ? "#fbbf24" : ok ? "#34d399" : "#f87171";
  const icon  = ok === "warn" ? "⚠️" : ok ? "✅" : "☐";
  return (
    <View style={{ flexDirection: "row", gap: 8, marginBottom: 6 }}>
      <Text style={{ fontSize: 14, lineHeight: 18 }}>{icon}</Text>
      <Text style={{ color: ok ? "white" : "#64748b", fontSize: 12, flex: 1 }}>{text}</Text>
    </View>
  );
}

BAGIAN 10: TOMBOL WATCHLIST
Ada di Verdict Card. POST/DELETE ke watchlist API:

async function toggleWatchlist(symbol, isAlreadyInWL) {
  const url = `${API}/api/watchlist${isAlreadyInWL ? `/${symbol}` : ""}`;
  await fetch(url, {
    method: isAlreadyInWL ? "DELETE" : "POST",
    headers: { "Content-Type": "application/json", Authorization: `Bearer ${jwt}` },
    body: isAlreadyInWL ? undefined : JSON.stringify({ symbol }),
  });
}
