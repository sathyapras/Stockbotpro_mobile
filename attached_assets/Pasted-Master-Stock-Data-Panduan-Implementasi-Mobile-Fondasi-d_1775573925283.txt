GAMBARAN UMUM
Master Stock adalah database utama seluruh saham IDX yang diupload dari Amibroker AFL Scanner setiap hari oleh admin. Satu file CSV berisi semua saham dengan harga, volume, FA ratios, sektor, dan index membership.

Dipakai oleh hampir semua screen:

Screen	Data yang diambil
Market Summary	Advancers · Decliners · Total Value
StockTools count badges	changePercent · vol50dPct
StockTools hasil screener	price · volume · sector · name
Stock Detail Financials	P/E · P/B · Beta · ROE · EPS · DY · D/E
Sector Rotation	Saham per sektor · top stock hari ini
Global Screener	Semua field untuk multi-filter
Bandar Detector	Sektor · indexCategory
Watchlist	Harga live untuk semua saham di watchlist

Kalkulasi Breadth di Frontend
// Jika perlu hitung manual dari masterData
function calcBreadth(stocks) {
  let advancers = 0, decliners = 0, unchanged = 0;
  let totalValue = 0;
  
  for (const s of stocks) {
    const chg = s.changePercent ?? 0;
    if (chg > 0.05)  advancers++;
    else if (chg < -0.05) decliners++;
    else unchanged++;
    totalValue += (s.value ?? 0) * 1e9;  // value dalam Bn Rp → konversi ke Rp
  }
  
  const total = advancers + decliners + unchanged;
  return {
    advancers, decliners, unchanged, total,
    totalValueT: (totalValue / 1e12).toFixed(1),  // dalam Triliun Rp
    advancerPct: Math.round((advancers / total) * 100),
    declinerPct: Math.round((decliners / total) * 100),
  };
}

// Semua filter dilakukan di sisi mobile, tidak perlu API tambahan
function filterStocks(stocks, filters) {
  let result = [...stocks];
  
  // Filter by index
  if (filters.index === "LQ45") {
    result = result.filter(s => (s.indexCategory ?? "").toUpperCase().includes("LQ45"));
  } else if (filters.index === "KOMPAS100") {
    result = result.filter(s => (s.indexCategory ?? "").toUpperCase().includes("KOMPAS"));
  } else if (filters.index === "JII30") {
    result = result.filter(s => (s.indexCategory ?? "").toUpperCase().includes("JII"));
  }
  
  // Filter by sector
  if (filters.sector) {
    result = result.filter(s =>
      (s.sector ?? "").toLowerCase().includes(filters.sector.toLowerCase())
    );
  }
  
  // Filter by search
  if (filters.search) {
    const q = filters.search.toUpperCase();
    result = result.filter(s =>
      s.symbol.includes(q) || (s.name ?? "").toUpperCase().includes(q)
    );
  }
  
  // Filter: hanya naik hari ini
  if (filters.onlyUp) {
    result = result.filter(s => (s.changePercent ?? 0) > 0);
  }
  
  // Filter: volume spike
  if (filters.volumeSpike) {
    result = result.filter(s => (s.vol50dPct ?? 0) >= 200);
  }
  
  return result;
}
// Sort options
function sortStocks(stocks, sortBy) {
  const sorted = [...stocks];
  switch (sortBy) {
    case "change_desc": return sorted.sort((a, b) => (b.changePercent ?? 0) - (a.changePercent ?? 0));
    case "change_asc":  return sorted.sort((a, b) => (a.changePercent ?? 0) - (b.changePercent ?? 0));
    case "volume_desc": return sorted.sort((a, b) => (b.volume ?? 0) - (a.volume ?? 0));
    case "value_desc":  return sorted.sort((a, b) => (b.value ?? 0) - (a.value ?? 0));
    case "cap_desc":    return sorted.sort((a, b) => (b.marketCap ?? 0) - (a.marketCap ?? 0));
    case "ret10d":      return sorted.sort((a, b) => (b.return10d ?? 0) - (a.return10d ?? 0));
    case "alpha":       return sorted.sort((a, b) => a.symbol.localeCompare(b.symbol));
    default:            return sorted;
  }
}

function StockCard({ stock, onPress }) {
  const isUp = (stock.changePercent ?? 0) >= 0;
  const color = isUp ? "#34d399" : "#f87171";
  
  // Index badges
  const cat = (stock.indexCategory ?? "").toUpperCase();
  const badges = [];
  if (cat.includes("LQ45"))     badges.push({ label: "LQ45",  color: "#60a5fa" });
  if (cat.includes("KOMPAS"))   badges.push({ label: "K100",  color: "#a78bfa" });
  if (cat.includes("JII"))      badges.push({ label: "JII",   color: "#34d399" });
  
  return (
    <TouchableOpacity onPress={() => onPress(stock.symbol)}
      style={{ padding: 12, borderRadius: 12, backgroundColor: "#1e2433", marginBottom: 8 }}>
      
      {/* Row 1: Ticker + Harga */}
      <View style={{ flexDirection: "row", justifyContent: "space-between" }}>
        <View style={{ flexDirection: "row", alignItems: "center", gap: 6 }}>
          <Text style={{ fontWeight: "900", fontSize: 15, color: "white" }}>{stock.symbol}</Text>
          {badges.map((b, i) => (
            <View key={i} style={{ paddingHorizontal: 4, paddingVertical: 1,
              borderRadius: 4, backgroundColor: b.color + "25" }}>
              <Text style={{ color: b.color, fontSize: 8, fontWeight: "700" }}>{b.label}</Text>
            </View>
          ))}
        </View>
        <Text style={{ fontWeight: "700", fontSize: 15, color: "white" }}>
          {stock.close?.toLocaleString("id-ID")}
        </Text>
      </View>
      
      {/* Row 2: Nama + % Change */}
      <View style={{ flexDirection: "row", justifyContent: "space-between", marginTop: 2 }}>
        <Text style={{ color: "#94a3b8", fontSize: 11 }} numberOfLines={1}>
          {stock.name}
        </Text>
        <Text style={{ color, fontWeight: "700", fontSize: 12 }}>
          {isUp ? "▲" : "▼"} {Math.abs(stock.changePercent ?? 0).toFixed(2)}%
        </Text>
      </View>

      <View style={{ flexDirection: "row", justifyContent: "space-between", marginTop: 4 }}>

        <Text style={{ color: "#60a5fa", fontSize: 10 }}>

            <Text style={{ color: "#fbbf24" }}> ★{Math.round(stock.vol50dPct ?? 0)}%</Text>

BAGIAN 5: SEKTOR SUMMARY DARI MASTER STOCK
Bangun summary per sektor langsung di client:

function buildSectorSummary(stocks) {
  const sectorMap = new Map();
  
  for (const s of stocks) {
    const sector = s.sector || "Lainnya";
    if (!sectorMap.has(sector)) {
      sectorMap.set(sector, { name: sector, stocks: [], advancers: 0, decliners: 0, totalValue: 0 });
    }
    const sec = sectorMap.get(sector);
    sec.stocks.push(s);
    if ((s.changePercent ?? 0) > 0) sec.advancers++;
    else if ((s.changePercent ?? 0) < 0) sec.decliners++;
    sec.totalValue += s.value ?? 0;  // Miliar Rp
  }
  
  return Array.from(sectorMap.values())
    .map(sec => ({
      name: sec.name,
      stockCount: sec.stocks.length,
      advancers: sec.advancers,
      decliners: sec.decliners,
      avgChange: sec.stocks.reduce((s, st) => s + (st.changePercent ?? 0), 0) / sec.stocks.length,
      totalValueBn: sec.totalValue,
      topStock: sec.stocks.sort((a, b) => (b.changePercent ?? 0) - (a.changePercent ?? 0))[0],
    }))
    .sort((a, b) => b.avgChange - a.avgChange);
}

function getVolumeLabel(vol50dPct) {
  if (vol50dPct >= 300) return { label: "MEGA SPIKE", color: "#a78bfa", show: true };

}

BAGIAN 9: HELPER FUNCTIONS
// Format volume
function formatVolume(vol) {
  if (vol >= 1e9) return `${(vol / 1e9).toFixed(1)}B`;
  if (vol >= 1e6) return `${(vol / 1e6).toFixed(1)}M`;
  if (vol >= 1e3) return `${(vol / 1e3).toFixed(0)}K`;
  return String(vol);
}
// Format Market Cap (dalam Miliar Rp dari master, convert ke Triliun/Miliar)
function formatMarketCap(bn) {
  if (bn >= 1000) return `Rp ${(bn / 1000).toFixed(1)}T`;
  return `Rp ${bn?.toFixed(0)}B`;
}
// Format nilai transaksi (dalam Miliar Rp)
function formatValue(bn) {
  if (bn >= 1000) return `${(bn / 1000).toFixed(1)}T`;
  return `${bn?.toFixed(1)}B`;
}
// Warna berdasar change%
function changeColor(pct) {
  if (pct > 0.5)  return "#34d399";  // hijau
  if (pct < -0.5) return "#f87171";  // merah
  return "#94a3b8";                   // abu (flat)
}
// Return% color dengan threshold berbeda
function returnColor(pct) {
  if (pct >= 3)  return "#34d399";
  if (pct >= 0)  return "#6ee7b7";
  if (pct >= -3) return "#fca5a5";
  return "#f87171";
}

function DataFreshnessChip({ uploadedAt }) {
  const now = new Date();
  const uploaded = new Date(uploadedAt);
  const hoursAgo = Math.floor((now - uploaded) / 3600000);
  
  let label, color;
  if (hoursAgo < 2)  { label = "Fresh";    color = "#34d399"; }
  else if (hoursAgo < 12) { label = `${hoursAgo}j lalu`; color = "#fbbf24"; }
  else               { label = "Perlu update"; color = "#f87171"; }
  
  return (
    <View style={{ flexDirection: "row", alignItems: "center", gap: 4 }}>
      <View style={{ width: 6, height: 6, borderRadius: 3, backgroundColor: color }} />
      <Text style={{ color, fontSize: 10 }}>Data: {label}</Text>
    </View>
  );
}
