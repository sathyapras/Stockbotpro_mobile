const SNAPSHOT_TABS = [
  { key: "acc",    icon: "⭐", label: "Top Akumulasi", badgeColor: "#16a34a" },
  { key: "entry",  icon: "🚀", label: "Entry Peluang",  badgeColor: "#0ea5e9" },
  { key: "warn",   icon: "⚠️", label: "Peringatan",     badgeColor: "#f97316" },
  { key: "trend",  icon: "✅", label: "Strong Trend",   badgeColor: "#a78bfa" },
];
function filterByTab(smData, tab) {
  switch (tab) {
    case "acc":
      return smData
        .filter(s => ["IGNITION", "EARLY_ACC", "STRONG_TREND"].includes(s.phase))
        .sort((a, b) => b.flowScore - a.flowScore)
        .slice(0, 8);
    case "entry":
      return smData
        .filter(s => s.phase === "IGNITION")
        .sort((a, b) => b.avg3d - a.avg3d)
        .slice(0, 8);
    case "warn":
      return smData
        .filter(s => ["EXHAUSTION", "DISTRIBUTION"].includes(s.phase))
        .sort((a, b) => a.flowScore - b.flowScore)  // worst first
        .slice(0, 8);
    case "trend":
      return smData
        .filter(s => s.phase === "STRONG_TREND")
        .sort((a, b) => b.avg3d - a.avg3d)
        .slice(0, 8);
    default:
      return [];
  }
}
// Badge label per phase
function getPhaseBadge(phase) {

    DISTRIBUTION: { label: "DIST", color: "#dc2626" },
    CHURNING:     { label: "NEU", color: "#475569" },
  };
  return map[phase] ?? { label: "NEU", color: "#475569" };
}

// Mapping phase → display label
const PHASE_DISPLAY = {
  IGNITION:     "Akumulasi Kuat",
  EARLY_ACC:    "Akumulasi",
  CHURNING:     "Netral",
  STRONG_TREND: "Akumulasi Kuat",  // gabung dengan IGNITION
  EXHAUSTION:   "Distribusi",
  DISTRIBUTION: "Distribusi Kuat",
};
const PHASE_COLORS = {
  "Akumulasi Kuat": "#10b981",
  "Akumulasi":      "#34d399",
  "Netral":         "#fbbf24",
  "Distribusi":     "#f87171",
  "Distribusi Kuat":"#dc2626",
};
function buildPhaseStats(smData) {
  const ORDER = ["Akumulasi Kuat", "Akumulasi", "Netral", "Distribusi", "Distribusi Kuat"];
  const counts = Object.fromEntries(ORDER.map(k => [k, 0]));
  let totalFlow = 0;
  for (const s of smData) {
    const label = PHASE_DISPLAY[s.phase] ?? "Netral";
    counts[label] = (counts[label] ?? 0) + 1;
    totalFlow += s.flowScore ?? 0;
  }
  const total = smData.length || 1;
  const avgFlow = Math.round(totalFlow / total);
  const accPct  = Math.round(((counts["Akumulasi Kuat"] + counts["Akumulasi"]) / total) * 100);
  const distPct = Math.round(((counts["Distribusi"] + counts["Distribusi Kuat"]) / total) * 100);
  return {
    phases: ORDER.map(label => ({
      label, count: counts[label],
      pct: Math.round((counts[label] / total) * 100),
      color: PHASE_COLORS[label],
    })),
    total, avgFlow, accPct, distPct,
  };
}

Logika Kalkulasi Risk Score (0-10)
function calcMarketRiskScore(sentiment) {
  const s = sentiment?.sentiment;
  if (!s) return { score: 5, components: [], label: "MEDIUM RISK", color: "#fbbf24" };
  let score = 0;
  const components = [];
  // ── VIX Fear Index ────────────────────────────────────────
  let vixPoin = 0;
  const vix = s.vix ?? 0;
  if      (vix >= 30) { vixPoin = 4; }
  else if (vix >= 25) { vixPoin = 3; }
  else if (vix >= 20) { vixPoin = 2; }
  else if (vix >= 15) { vixPoin = 1; }
  score += vixPoin;
  components.push({
    icon: "😱", label: "VIX Fear Index",
    poin: vixPoin > 0 ? `+${vixPoin} poin` : "aman",
    desc: `VIX ${vix.toFixed(1)} — ${s.fearLabel?.toLowerCase()}`,
    color: vixPoin >= 3 ? "#f87171" : vixPoin >= 2 ? "#fbbf24" : "#34d399",
  });
  // ── Dollar Index (DXY) ────────────────────────────────────
  let dxyPoin = 0;
  const dxy = s.dxyValue ?? 0;
  if      (dxy >= 105) { dxyPoin = 2; }
  else if (dxy >= 102) { dxyPoin = 1; }
  score += dxyPoin;
  components.push({
    icon: "💵", label: "Dollar Index (DXY)",
    poin: dxyPoin > 0 ? `+${dxyPoin} poin` : "aman",
    desc: `DXY ${dxy.toFixed(1)} — ${s.dxyBias}`,
    color: dxyPoin >= 2 ? "#f87171" : dxyPoin >= 1 ? "#fbbf24" : "#34d399",
  });
  // ── IHSG vs MA20 ─────────────────────────────────────────
  let ma20Poin = 0;
  if (!s.ihsgAboveMA20) {
    const gap = s.ihsgMA20 ? ((s.ihsgMA20 - s.ihsgClose) / s.ihsgMA20 * 100) : 0;
    ma20Poin = gap >= 5 ? 3 : gap >= 2 ? 2 : 1;
  }
  score += ma20Poin;
  components.push({
    icon: "📉", label: "IHSG vs MA20",
    poin: ma20Poin > 0 ? `+${ma20Poin} poin` : "aman",
    desc: s.ihsgAboveMA20
      ? `IHSG ${s.ihsgClose?.toLocaleString("id-ID")} di atas MA20 — tren teknis sehat`
      : `IHSG ${s.ihsgClose?.toLocaleString("id-ID")} | MA20 ${s.ihsgMA20?.toLocaleString("id-ID")} — tren teknis sedang rusak`,
    color: ma20Poin >= 2 ? "#f87171" : ma20Poin >= 1 ? "#fbbf24" : "#34d399",
  });
  // ── USD/IDR ───────────────────────────────────────────────
  let idrPoin = 0;
  const usdIdr = s.usdIdr ?? 0;
  if      (usdIdr >= 17000) { idrPoin = 3; }
  else if (usdIdr >= 16500) { idrPoin = 2; }
  else if (usdIdr >= 16000) { idrPoin = 1; }
  score += idrPoin;
  components.push({
    icon: "🌏", label: "Nilai Tukar USD/IDR",
    poin: idrPoin > 0 ? `+${idrPoin} poin` : "aman",
    desc: `USD/IDR ${usdIdr.toLocaleString("id-ID")} — ${s.usdIdrContext}`,
    color: idrPoin >= 2 ? "#f87171" : idrPoin >= 1 ? "#fbbf24" : "#34d399",
  });
  // Cap at 10
  const finalScore = Math.min(10, score);
  let label, color, bg, strategy;
  if (finalScore >= 8)      { label = "HIGH RISK";    color = "#f87171"; bg = "#2d0a0a"; strategy = highRiskStrategy(); }
  else if (finalScore >= 6) { label = "MEDIUM-HIGH";  color = "#f97316"; bg = "#1c0a00"; strategy = medHighStrategy(); }
  else if (finalScore >= 4) { label = "MEDIUM RISK";  color = "#fbbf24"; bg = "#1c1500"; strategy = medStrategy(); }
  else                      { label = "LOW RISK";     color = "#34d399"; bg = "#052e16"; strategy = lowStrategy(); }
  return { score: finalScore, components, label, color, bg, strategy };
}
