// Fungsi interpretasi VWAP (server sudah kirim bandarAvg1d)
function vwapLine(currentPrice, bandarAvg1d) {
  if (!bandarAvg1d || !currentPrice) return null;
  const diff = currentPrice - bandarAvg1d;
  const pct  = ((diff / bandarAvg1d) * 100).toFixed(1);
  
  if (diff > 0) return {
    text: `Harga ${diff.toLocaleString("id-ID")} di ATAS VWAP bandar (+${pct}%)`,
    color: "#34d399",
    icon: "📈",
  };
  return {
    text: `Harga ${Math.abs(diff).toLocaleString("id-ID")} di BAWAH VWAP bandar (${pct}%)`,
    color: "#f87171",
    icon: "📉",
  };
}

Fix 4: Signal 5D dan 10D — Fallback dari Nilai NBS

function deriveSignal(rawCtx, nbsValue) {
  // 1. Pakai context dari server jika ada
  if (rawCtx && rawCtx.trim() !== "") {
    return flowToSignal(rawCtx);
  }
  // 2. Fallback: derive dari nilai NBS
  if (nbsValue > 0)  return "Accumulation";
  if (nbsValue < 0)  return "Distribution";
  return "Neutral";
}

      </Text>

// Strength label:
function getBandarStrength(s1, s5, s10) {
  const signals = [s1, s5, s10];
  const acc  = signals.filter(s => s === "Accumulation").length;
  const dist = signals.filter(s => s === "Distribution").length;
  if (acc === 3)  return { label: "ACC KUAT ✦", color: "#10b981" };
  if (acc === 2)  return { label: "ACC",         color: "#34d399" };
  if (dist === 3) return { label: "DIST KUAT ✦", color: "#dc2626" };
  if (dist === 2) return { label: "DIST",        color: "#f87171" };
  return           { label: "MIXED",             color: "#64748b" };
}

function getMarketPulse(smBrokPulse, smAccPct) {
  // Hybrid: 70% scanner signal + 30% broker pulse
  const hybridPulse = Math.round(smBrokPulse * 0.7 + smAccPct * 0.3);
  
  let label, color, bg, desc;
  if (hybridPulse >= 70) {
    label = "BULLISH";   color = "#34d399"; bg = "#052e16";
    desc  = "Kondisi pasar kuat — momentum mendukung";
  } else if (hybridPulse >= 50) {
    label = "NETRAL";    color = "#fbbf24"; bg = "#1c1500";
    desc  = "Pasar campuran — selektif dan hati-hati";
  } else if (hybridPulse >= 35) {
    label = "WASPADA";   color = "#f97316"; bg = "#1c0a00";
    desc  = "Momentum lemah — lebih banyak cash";
  } else {
    label = "BEARISH";   color = "#f87171"; bg = "#2d0a0a";
    desc  = "Kondisi buruk — hindari posisi baru";
  }
  return { hybridPulse, label, color, bg, desc };
}

// Hitung phase distribution dari smart-money endpoint
// GET /api/broker-summary/smart-money → { data: [...] }
// Setiap item punya field: phase
const PHASE_MAP = {
  IGNITION:     { label: "Markup",       color: "#60a5fa" },
  EARLY_ACC:    { label: "Accumulation", color: "#34d399" },
  STRONG_TREND: { label: "Markup",       color: "#60a5fa" },
  EXHAUSTION:   { label: "Late Stage",   color: "#fbbf24" },
  DISTRIBUTION: { label: "Distribution", color: "#f87171" },
  CHURNING:     { label: "Consolidation",color: "#94a3b8" },
};
function buildPhaseDistribution(smartMoneyData) {
  const counts = {};
  const total = smartMoneyData.length;
  
  for (const stock of smartMoneyData) {
    const phase = stock.phase ?? "CHURNING";
    const display = PHASE_MAP[phase]?.label ?? "Consolidation";
    counts[display] = (counts[display] ?? 0) + 1;
  }
  
  return Object.entries(counts)
    .map(([label, count]) => ({
      label,
      count,
      pct: Math.round((count / total) * 100),
      color: Object.values(PHASE_MAP).find(p => p.label === label)?.color ?? "#94a3b8",
    }))
    .sort((a, b) => b.count - a.count);
}

function buildFlowStateDistribution(bandarList) {
  const counts = {
    "STRONG ACC":  0,
    "ACC":         0,
    "NEUTRAL":     0,
    "DIST":        0,
    "STRONG DIST": 0,
  };
  
  for (const item of bandarList) {
    const fs = (item.flowState ?? "").toUpperCase();
    if      (fs.includes("STRONG ACCUMULATION"))  counts["STRONG ACC"]++;
    else if (fs.includes("ACCUMULATION"))          counts["ACC"]++;
    else if (fs.includes("STRONG DISTRIBUTION"))   counts["STRONG DIST"]++;
    else if (fs.includes("DISTRIBUTION"))          counts["DIST"]++;
    else                                            counts["NEUTRAL"]++;
  }
  
  const total = bandarList.length || 1;
  return [
    { key: "STRONG ACC",  label: "★ Akumulasi",  color: "#10b981", icon: "★" },
    { key: "ACC",         label: "Akumulasi",    color: "#34d399", icon: "●" },
    { key: "NEUTRAL",     label: "Netral",       color: "#64748b", icon: "○" },
    { key: "DIST",        label: "Distribusi",   color: "#f87171", icon: "●" },
    { key: "STRONG DIST", label: "★ Distribusi", color: "#dc2626", icon: "★" },
  ].map(item => ({
    ...item,
    count: counts[item.key],
    pct: Math.round((counts[item.key] / total) * 100),
  }));
}

// Hitung total NBS dari bandarList (sudah ada nbs data)
// bandarList = GET /api/analysis/bandar-detector?index=COMPOSITE
function buildMarketFlow(bandarList) {
  let total5d = 0, total10d = 0;
  let accCount5d = 0;
  
  for (const item of bandarList) {
    total5d  += item.fiveDayBillion  ?? 0;
    total10d += item.tenDayBillion   ?? 0;
    if ((item.fiveDayBillion ?? 0) > 0) accCount5d++;
  }
  
  const formatT = (bn) => {
    const abs  = Math.abs(bn);
    const sign = bn >= 0 ? "+" : "-";
    if (abs >= 1000) return `${sign}${(abs / 1000).toFixed(1)}T`;
    return `${sign}${abs.toFixed(0)}B`;
  };
  
  return {
    nbs5d:   total5d,
    nbs10d:  total10d,
    nbs5dStr:  formatT(total5d),
    nbs10dStr: formatT(total10d),
    label5d:   total5d >= 0 ? "Net beli (akumulasi)" : "Net jual (distribusi)",
    label10d:  total10d >= 0 ? "Net beli (akumulasi)" : "Net jual (distribusi)",
    color5d:   total5d >= 0 ? "#34d399" : "#f87171",
    color10d:  total10d >= 0 ? "#34d399" : "#f87171",
  };
}

        <Text style={{ color: "#475569", fontSize: 11 }}>
          {new Date().toLocaleDateString("id-ID", { weekday: "short", day: "numeric", month: "short", year: "numeric" })}
        </Text>
      </View>
      
      {/* 4 Section Cards */}
      {aggregate && <MarketPulseCard aggregate={aggregate} />}
      {smData?.length > 0 && <PhaseDistributionCard smartMoneyData={smData} />}
      {bandarList.length > 0 && <FlowStateCard bandarList={bandarList} />}
      {bandarList.length > 0 && <SmartMoneyFlowCard bandarList={bandarList} />}
      
    </ScrollView>
  );
}
