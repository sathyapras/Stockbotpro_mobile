SETUP DEVICE ID
Panggil sekali saat screen mount, simpan di state lokal.

import * as SecureStore from "expo-secure-store";
import "react-native-get-random-values";
import { v4 as uuidv4 } from "uuid";
const DEVICE_KEY = "sbp_device_id";
async function getOrCreateDeviceId(): Promise<string> {
  let id = await SecureStore.getItemAsync(DEVICE_KEY);
  if (!id) {
    id = uuidv4();
    await SecureStore.setItemAsync(DEVICE_KEY, id);
  }
  return id;
}

    setDeviceId(id);

async function fetchAll(id: string) {
  setLoading(true);
  try {
    const [tRes, sRes] = await Promise.all([
      fetch(`${BASE}/trading-log?device_id=${id}`),
      fetch(`${BASE}/trading-log/summary?device_id=${id}`),
    ]);
    const tJson = await tRes.json();
    const sJson = await sRes.json();
    if (tJson.ok) setTrades(tJson.data);
    if (sJson.ok) setSummary(sJson);
  } catch {
    // tampilkan empty state
  } finally {
    setLoading(false);
  }
}

function TradeCard({ trade, onClose, onDelete }) {
  const entry   = Number(trade.entryPrice);
  const exit    = trade.exitPrice ? Number(trade.exitPrice) : null;
  const pnlPct  = exit != null ? ((exit - entry) / entry * 100) : null;
  const isOpen  = trade.status === "OPEN";
  // Hitung hold days
  const entryMs = new Date(trade.entryDate).getTime();
  const endDate = trade.exitDate ? new Date(trade.exitDate) : new Date();
  const holdDays = Math.round((endDate.getTime() - entryMs) / 86_400_000);
  return (
    <View style={styles.card}>
      {/* ── Header: Ticker + Badge signal + Status ── */}
      <View style={styles.cardHeader}>
        <View style={{ flexDirection: "row", alignItems: "center", gap: 8 }}>
          <Text style={styles.ticker}>{trade.ticker}</Text>
          <View style={[styles.signalBadge,
            { backgroundColor: trade.signalType === "BOW"
              ? "rgba(34,197,94,0.15)" : "rgba(59,130,246,0.15)" }]}>
            <Text style={[styles.signalText,
              { color: trade.signalType === "BOW" ? "#22c55e" : "#3b82f6" }]}>
              {trade.signalType}
            </Text>
          </View>
        </View>
        {/* Status badge */}
        <View style={[styles.statusBadge,
          { backgroundColor: isOpen
            ? "rgba(245,158,11,0.15)" : "rgba(100,116,139,0.15)" }]}>
          <Text style={[styles.statusText,
            { color: isOpen ? "#f59e0b" : "#64748b" }]}>
            {isOpen ? "OPEN" : "CLOSED"}
          </Text>
        </View>
      </View>
      {/* ── Grid harga: Entry / SL / TP1 / TP2 / Exit ── */}
      <View style={styles.priceGrid}>
        <PriceItem label="Entry" value={formatPrice(trade.entryPrice)} />
        <PriceItem label="SL"    value={formatPrice(trade.slPrice)}    color="#ef4444" />
        <PriceItem label="TP1"   value={formatPrice(trade.tp1Price)}   color="#22c55e" />
        <PriceItem label="TP2"   value={formatPrice(trade.tp2Price)}   color="#22c55e" />
        {exit != null && (
          <PriceItem label="Exit" value={formatPrice(trade.exitPrice!)}
            color={pnlPct! >= 0 ? "#22c55e" : "#ef4444"} />
        )}
      </View>
      {/* ── P&L dan Hold Days ── */}
      <View style={styles.cardFooter}>
        <View style={{ flexDirection: "row", gap: 12, alignItems: "center" }}>
          {pnlPct != null && (
            <Text style={[styles.pnl, { color: pnlPct >= 0 ? "#22c55e" : "#ef4444" }]}>
              {pnlPct >= 0 ? "+" : ""}{pnlPct.toFixed(2)}%
            </Text>
          )}
          <Text style={styles.holdDays}>
            📅 {trade.entryDate.slice(5).replace("-", "/")}
            {trade.exitDate ? ` → ${trade.exitDate.slice(5).replace("-", "/")}` : ""}
            {" "}·{" "}{holdDays}h
          </Text>
        </View>
        {/* Action buttons */}
        <View style={{ flexDirection: "row", gap: 8 }}>
          {isOpen && (
            <Pressable style={styles.btnClose} onPress={() => onClose(trade)}>
              <Text style={styles.btnCloseText}>Tutup</Text>
            </Pressable>
          )}
          <Pressable style={styles.btnDelete} onPress={() => onDelete(trade)}>
            <Text style={styles.btnDeleteText}>🗑</Text>
          </Pressable>
        </View>
      </View>
      {/* ── Catatan (jika ada) ── */}
      {trade.notes ? (
        <Text style={styles.notes}>📝 {trade.notes}</Text>
      ) : null}
    </View>
  );
}
function PriceItem({ label, value, color = "#94a3b8" }) {
  return (
    <View style={{ alignItems: "center" }}>
      <Text style={[styles.priceValue, { color }]}>{value}</Text>
      <Text style={styles.priceLabel}>{label}</Text>
    </View>
  );
}
function formatPrice(val: string | number): string {
  const n = Number(val);
  if (n >= 1000) return (n / 1000).toFixed(1) + "k";
  return String(Math.round(n));
}

function ModalTambah({ visible, deviceId, onClose, onSuccess }) {
  const [ticker,     setTicker]     = useState("");
  const [signalType, setSignalType] = useState<"BOW" | "BOS">("BOW");
  const [entry,      setEntry]      = useState("");
  const [sl,         setSl]         = useState("");
  const [tp1,        setTp1]        = useState("");
  const [tp2,        setTp2]        = useState("");
  const [date,       setDate]       = useState(todayStr()); // "YYYY-MM-DD"
  const [notes,      setNotes]      = useState("");
  const [saving,     setSaving]     = useState(false);
  async function handleSave() {
    if (!ticker.trim() || !entry || !sl || !tp1 || !tp2 || !date) {
      Alert.alert("Lengkapi semua field wajib");
      return;
    }
    setSaving(true);
    try {
      const res = await fetch("https://stockbotpro.replit.app/api/trading-log", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          device_id:   deviceId,
          ticker:      ticker.trim().toUpperCase(),
          signal_type: signalType,
          entry_price: Number(entry),
          sl_price:    Number(sl),
          tp1_price:   Number(tp1),
          tp2_price:   Number(tp2),
          entry_date:  date,
          notes:       notes.trim() || undefined,
        }),
      });
      const json = await res.json();
      if (!json.ok) throw new Error(json.error);
      onSuccess();
      onClose();
    } catch (e: any) {
      Alert.alert("Gagal", e.message);
    } finally {
      setSaving(false);
    }
  }
  return (
    <Modal visible={visible} animationType="slide" transparent>
      <View style={styles.overlay}>
        <View style={styles.sheet}>
          <Text style={styles.sheetTitle}>Tambah Trade</Text>
          {/* Toggle BOW / BOS */}
          <View style={styles.toggleRow}>
            {(["BOW","BOS"] as const).map((s) => (
              <Pressable key={s}
                style={[styles.toggleBtn,
                  signalType === s && styles.toggleBtnActive,
                  { borderColor: s === "BOW" ? "#22c55e" : "#3b82f6" },
                  signalType === s && { backgroundColor:
                    s === "BOW" ? "rgba(34,197,94,0.2)" : "rgba(59,130,246,0.2)" }
                ]}
                onPress={() => setSignalType(s)}>
                <Text style={[styles.toggleText,
                  { color: signalType === s
                    ? (s === "BOW" ? "#22c55e" : "#3b82f6")
                    : "#64748b" }]}>
                  {s === "BOW" ? "🟢 BOW" : "🔵 BOS"}
                </Text>
              </Pressable>
            ))}
          </View>
          {/* Field input */}
          <TextInput style={styles.input} placeholder="Ticker (misal: BBRI)"
            placeholderTextColor="#475569" value={ticker}
            onChangeText={(v) => setTicker(v.toUpperCase())}
            autoCapitalize="characters" />
          <View style={styles.row2}>
            <TextInput style={[styles.input, { flex: 1 }]} placeholder="Entry"
              placeholderTextColor="#475569" keyboardType="decimal-pad"
              value={entry} onChangeText={setEntry} />
            <TextInput style={[styles.input, { flex: 1 }]} placeholder="Stop Loss"
              placeholderTextColor="#475569" keyboardType="decimal-pad"
              value={sl} onChangeText={setSl} />
          </View>
          <View style={styles.row2}>
            <TextInput style={[styles.input, { flex: 1 }]} placeholder="TP1"
              placeholderTextColor="#475569" keyboardType="decimal-pad"
              value={tp1} onChangeText={setTp1} />
            <TextInput style={[styles.input, { flex: 1 }]} placeholder="TP2"
              placeholderTextColor="#475569" keyboardType="decimal-pad"
              value={tp2} onChangeText={setTp2} />
          </View>
          {/* Tanggal — bisa pakai DateTimePicker atau input manual */}
          <TextInput style={styles.input} placeholder="Tanggal (YYYY-MM-DD)"
            placeholderTextColor="#475569" value={date} onChangeText={setDate} />
          <TextInput style={[styles.input, { height: 64, textAlignVertical: "top" }]}
            placeholder="Catatan (opsional)" placeholderTextColor="#475569"
            multiline value={notes} onChangeText={setNotes} />
          <View style={styles.row2}>
            <Pressable style={[styles.btn, styles.btnCancel]} onPress={onClose}>
              <Text style={styles.btnCancelText}>Batal</Text>
            </Pressable>
            <Pressable style={[styles.btn, styles.btnSave]} onPress={handleSave}
              disabled={saving}>
              <Text style={styles.btnSaveText}>
                {saving ? "Menyimpan..." : "Simpan Trade"}
              </Text>
            </Pressable>
          </View>
        </View>
      </View>
    </Modal>
  );
}
function todayStr(): string {
  return new Date().toISOString().slice(0, 10);
}

MODAL: Tutup Posisi
Preview P&L real-time berubah saat user mengetik exit price.

function ModalTutup({ visible, trade, deviceId, onClose, onSuccess }) {
  const [exitPrice, setExitPrice] = useState("");
  const [exitDate,  setExitDate]  = useState(todayStr());
  const [notes,     setNotes]     = useState("");
  const [saving,    setSaving]    = useState(false);
  // Preview P&L real-time
  const entry   = Number(trade?.entryPrice ?? 0);
  const exit    = Number(exitPrice) || 0;
  const pnlPct  = entry > 0 && exit > 0
    ? ((exit - entry) / entry * 100)
    : null;
  async function handleConfirm() {
    if (!exitPrice || !exitDate) {
      Alert.alert("Lengkapi exit price dan tanggal");
      return;
    }
    setSaving(true);
    try {
      const res = await fetch(
        `https://stockbotpro.replit.app/api/trading-log/${trade.id}/close?device_id=${deviceId}`,
        {
          method: "PATCH",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            exit_price: Number(exitPrice),
            exit_date:  exitDate,
            notes:      notes.trim() || undefined,
          }),
        }
      );
      const json = await res.json();
      if (!json.ok) throw new Error(json.error);
      onSuccess();
      onClose();
    } catch (e: any) {
      Alert.alert("Gagal", e.message);
    } finally {
      setSaving(false);
    }
  }
  if (!trade) return null;
  return (
    <Modal visible={visible} animationType="slide" transparent>
      <View style={styles.overlay}>
        <View style={styles.sheet}>
          <Text style={styles.sheetTitle}>Tutup Posisi</Text>
          <Text style={styles.sheetSubtitle}>{trade.ticker} · {trade.signalType}</Text>
          {/* Preview P&L real-time */}
          <View style={styles.pnlPreview}>
            <Text style={styles.pnlPreviewLabel}>Entry</Text>
            <Text style={styles.pnlPreviewValue}>{formatPrice(trade.entryPrice)}</Text>
            <Text style={styles.pnlArrow}>→</Text>
            <Text style={[styles.pnlPreviewValue,
              { color: exit === 0 ? "#94a3b8"
                : pnlPct! >= 0 ? "#22c55e" : "#ef4444" }]}>
              {exit > 0 ? formatPrice(exitPrice) : "—"}
            </Text>
            {pnlPct != null && (
              <View style={[styles.pnlBadge,
                { backgroundColor: pnlPct >= 0 ? "rgba(34,197,94,0.15)" : "rgba(239,68,68,0.15)" }]}>
                <Text style={{ color: pnlPct >= 0 ? "#22c55e" : "#ef4444",
                  fontWeight: "800", fontSize: 16 }}>
                  {pnlPct >= 0 ? "+" : ""}{pnlPct.toFixed(2)}%
                </Text>
              </View>
            )}
          </View>
          <TextInput style={styles.input} placeholder="Exit Price"
            placeholderTextColor="#475569" keyboardType="decimal-pad"
            value={exitPrice} onChangeText={setExitPrice} />
          <TextInput style={styles.input} placeholder="Tanggal Exit (YYYY-MM-DD)"
            placeholderTextColor="#475569" value={exitDate}
            onChangeText={setExitDate} />
          <TextInput style={[styles.input, { height: 56, textAlignVertical: "top" }]}
            placeholder="Catatan (opsional)" placeholderTextColor="#475569"
            multiline value={notes} onChangeText={setNotes} />
          <View style={styles.row2}>
            <Pressable style={[styles.btn, styles.btnCancel]} onPress={onClose}>
              <Text style={styles.btnCancelText}>Batal</Text>
            </Pressable>
            <Pressable style={[styles.btn, styles.btnSave]} onPress={handleConfirm}
              disabled={saving}>
              <Text style={styles.btnSaveText}>
                {saving ? "Menyimpan..." : "Konfirmasi Tutup"}
              </Text>
            </Pressable>
          </View>
        </View>
      </View>
    </Modal>
  );
}

HAPUS DENGAN KONFIRMASI DIALOG
function handleDeletePress(trade: Trade) {
  Alert.alert(
    "Hapus Trade",
    `Yakin ingin menghapus trade ${trade.ticker}?`,
    [
      { text: "Batal", style: "cancel" },
      {
        text: "Hapus",
        style: "destructive",
        onPress: async () => {
          try {
            const res = await fetch(
              `${BASE}/trading-log/${trade.id}?device_id=${deviceId}`,
              { method: "DELETE" }
            );
            const json = await res.json();
            if (!json.ok) throw new Error(json.error);
            await fetchAll(deviceId); // refresh list
          } catch (e: any) {
            Alert.alert("Gagal hapus", e.message);
          }
        },
      },
    ]
  );
}

              onDelete={handleDeletePress}
            />
          )}
          contentContainerStyle={{ paddingBottom: 32 }}
          showsVerticalScrollIndicator={false}
          refreshing={loading}
          onRefresh={() => fetchAll(deviceId)}
        />
      )}
      {/* Modals */}
      <ModalTambah
        visible={showTambah}
        deviceId={deviceId}
        onClose={() => setShowTambah(false)}
        onSuccess={() => fetchAll(deviceId)}
      />
      <ModalTutup
        visible={showTutup}
        trade={selectedTrade}
        deviceId={deviceId}
        onClose={() => { setShowTutup(false); setSelectedTrade(null); }}
        onSuccess={() => fetchAll(deviceId)}
      />
    </SafeAreaView>
  );
}
