Jejak Cuan — Panduan Implementasi Mobile
Mengganti Trade Log Screen menjadi Jejak Cuan
Screen ini menggantikan tampilan lama "Trade Log" (jurnal trade personal per device)
menjadi Jejak Cuan — rekam jejak publik sinyal BOW & BOS dari admin, bukan catatan pribadi user.

PERBEDAAN UTAMA: Trade Log vs Jejak Cuan
Trade Log (lama)	Jejak Cuan (baru)
Sumber data	Input manual oleh user per device	Sinyal resmi BOW & BOS dari admin
Akses	Private (per deviceId)	Closed: publik · Open: login required
Tombol "+ Tambah"	Ada — user mencatat sendiri	Hapus — data otomatis dari server
P&L	Hitung dari entry/exit user	Langsung dari field glPct server
Endpoint	/api/trading-log	/api/jejak-cuan
Filter	SEMUA / OPEN / CLOSED	SEMUA / BOS / BOW
ARSITEKTUR SCREEN
JejakCuanScreen
├── Header                — judul + subtitle + refresh button
├── FilterTabs            — SEMUA / BOS / BOW  
├── StatsBar              — Win Rate · Total Trade · Avg P&L · Avg Hold
├── MonthlyChart          — bar chart P&L bulanan (opsional, pakai Victory Native)
├── OpenSection           — posisi sedang berjalan (auth gate)
│   ├── AuthGate          — tampil Lock + tombol Login jika belum login
│   └── FlatList<OpenRow> — list posisi open (hanya untuk user login)
├── ClosedSection         — riwayat posisi closed (publik)
│   └── FlatList<ClosedRow>
├── FooterNote            — keterangan + timestamp update
└── EmptyState            — saat belum ada data sama sekali

TIPE DATA
type SignalFilter = "SEMUA" | "BOS" | "BOW";
interface Stats {
  total: number;
  wins: number;
  losses: number;
  winRate: number;
  avgPnl: number;
  totalPnl: number;
  avgHoldDays: number;
}
interface MonthlyItem {
  month: string;      // "2026-03"
  label: string;      // "Mar '26"
  pnl: number;        // total P&L bulan tsb (%)
  winRate: number;
  total: number;
  wins: number;
}
interface ClosedTrade {
  symbol: string;
  date: string;
  signalType: "BOS" | "BOW";
  glPct: number;
  holdDays: number;
  entry: number;
  exitPrice: number;
  exitReason: string;
  trend: string;
  win: boolean;
}
interface OpenTrade {
  symbol: string;
  date: string;
  signalType: "BOS" | "BOW";
  signal: string;     // "HOLD" | "BUY"
  glPct: number;
  holdDays: number;
  entry: number;
  close: number;
  stopLoss: number;
  tp1: number;
  tp2: number;
  hold: string;       // "7 hari"
  slPhase: string;    // "TGT1 HIT Trail ON"
  trend: string;
}
interface JejakData {
  stats: { combined: Stats; bos: Stats; bow: Stats };
  monthly: MonthlyItem[];
  closed: ClosedTrade[];
  meta: { bosUploadedAt: string | null; bowUploadedAt: string | null };
}

LOAD DATA SAAT MOUNT
useEffect(() => {
  loadToken();
  fetchPublic();
}, []);
async function loadToken() {
  const t = await SecureStore.getItemAsync("stockbot-auth-token");
  setToken(t);
  if (t) fetchOpen(t);
}
async function fetchPublic() {
  setLoading(true);
  setError(null);
  try {
    const res  = await fetch(`${BASE}/jejak-cuan`);
    const json = await res.json();
    if (json.ok) setData(json);
    else throw new Error(json.error ?? "Gagal memuat");
  } catch (e: any) {
    setError(e.message);
  } finally {
    setLoading(false);
  }
}
async function fetchOpen(t: string) {
  setOpenLoading(true);
  try {
    const res  = await fetch(`${BASE}/jejak-cuan/open`, {
      headers: { Authorization: `Bearer ${t}` },
    });
    if (res.status === 401) { setOpenData([]); return; }
    const json = await res.json();
    if (json.ok) setOpenData(json.open ?? []);
  } catch {
    setOpenData([]);
  } finally {
    setOpenLoading(false);
  }
}
function handleRefresh() {
  fetchPublic();
  if (token) fetchOpen(token);
}

function OpenSection({ open, loading, token, onGoLogin }) {
  return (
    <View style={styles.section}>
      {/* ── Header seksi ── */}
      <View style={styles.sectionHeader}>
        <Text style={styles.sectionIcon}>⚡</Text>
        <Text style={styles.sectionTitle}>Posisi Sedang Berjalan</Text>
        {token && (
          <Text style={styles.sectionCount}>{open.length} posisi</Text>
        )}
      </View>
      {/* ── Konten: Auth Gate atau list ── */}
      {!token ? (
        // Belum login → tampilkan Lock screen
        <View style={styles.authGate}>
          <Text style={styles.lockIcon}>🔒</Text>
          <Text style={styles.authTitle}>Login untuk melihat posisi terbuka</Text>
          <Text style={styles.authDesc}>
            Data posisi aktif hanya tersedia untuk member yang sudah register
          </Text>
          <Pressable style={styles.loginBtn} onPress={onGoLogin}>
            <Text style={styles.loginBtnText}>Login / Daftar →</Text>
          </Pressable>
        </View>
      ) : loading ? (
        <ActivityIndicator color="#00d4ff" style={{ marginVertical: 16 }} />
      ) : open.length === 0 ? (
        <Text style={styles.emptyText}>Tidak ada posisi terbuka saat ini</Text>
      ) : (
        open.map((t, i) => <OpenRow key={i} trade={t} />)
      )}
    </View>
  );
}

function OpenRow({ trade }: { trade: OpenTrade }) {
  const gl         = trade.glPct;
  const isPhaseHit = trade.slPhase?.toUpperCase().includes("TGT");
  return (
    <View style={styles.tradeRow}>
      {/* Left icon */}
      <View style={[styles.rowIcon, { backgroundColor: "rgba(245,158,11,0.15)" }]}>
        <Text style={{ fontSize: 13 }}>⚡</Text>
      </View>
      {/* Info tengah */}
      <View style={{ flex: 1 }}>
        <View style={{ flexDirection: "row", alignItems: "center", gap: 6, flexWrap: "wrap" }}>
          <Text style={styles.symbol}>{trade.symbol}</Text>
          <SignalBadge type={trade.signalType} />
          {isPhaseHit && (
            <View style={styles.phaseHitBadge}>
              <Text style={styles.phaseHitText}>🎯 Target Hit</Text>
            </View>
          )}
        </View>
        <Text style={styles.tradeSubText}>
          Entry {trade.entry.toLocaleString("id-ID")} · {trade.hold || `${trade.holdDays} hari`}
        </Text>
      </View>
      {/* P&L kanan */}
      <Text style={[styles.pnlText, { color: gl > 0 ? "#22c55e" : gl < 0 ? "#ef4444" : "#94a3b8" }]}>
        {gl > 0 ? "+" : ""}{gl.toFixed(1)}%
      </Text>
    </View>
  );
}
